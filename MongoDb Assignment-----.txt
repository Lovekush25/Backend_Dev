// 1. Find all students with GPA between 3.0 and 3.5
db.student.find({ gpa:{$gte: 3.0, $lte: 3.5}})

// 2. Find students enrolled in more than 5 courses (courses as array)
db.student.find({ courses: { $size: 6 } })   // OR better:
db.student.find({ $expr: { $gt: [ { $size: "$courses" }, 5 ]}})

// 3. Get top 10 students by GPA
db.student.find().sort({ gpa: -1 }).limit(10)

// 4. Count students by city
db.student.aggregate([{$group:{_id:"$city",count:{$sum:1}}}])

Exercise 4--->


// 1. Calculate average GPA by department
db.students.aggregate([{ $group: { _id: "$department", avgGPA: {$avg:"$gpa"}}}])
// 2. Find most popular courses (courses as array)
db.students.aggregate([
  { $unwind: "$courses" },
  { $group: { _id: "$courses", count: { $sum: 1 } } },
  { $sort: { count: -1 } }
])
// 3. Generate student performance report
db.students.aggregate([
  { $project: { name: 1, email: 1, gpa: 1,
      performance: {
        $cond: [
          { $gte: ["$gpa", 8] }, "Excellent",
          { $cond: [{ $gte: ["$gpa", 6] }, "Good", "Average"] }
        ]
      }
  }}
])
